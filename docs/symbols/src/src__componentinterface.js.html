 16 (function(){
 17   /**
 18    * @class
 19    * @name aw.components.utils.ComponentInterface
 20    * @param $scope
 21    * @constructor
 22    */
 23   function ComponentInterface($scope){
 24     /**
 25      * Child components
 26      * @type {aw.components.Component[]}
 27      * @instance
 28      */
 29     this.children = [];
 30     /**
 31      * Parent component
 32      * @type {aw.components.Component}
 33      * @instance
 34      */
 35     this.parent = null;
 36     /**
 37      * Executed when child component added. At this stage child does not know about parent component and its just finished initialization.
 38      * @property
 39      * @name aw.components.utils.ComponentInterface#childAdded
 40      * @type {aw.events.EventListener}
 41      * @instance
 42      */
 43     this.childAdded = this.createListener();
 44     /**
 45      * Executed when child component removed
 46      * @property
 47      * @name aw.components.utils.ComponentInterface#childRemoved
 48      * @type {aw.events.EventListener}
 49      * @instance
 50      */
 51     this.childRemoved = this.createListener();
 52     /**
 53      * Executed after component being added to the parent component.
 54      * At this point, parent was notified about new child and probably
 55      * already placed event handlers.
 56      * @property
 57      * @name aw.components.utils.ComponentInterface#addedToParent
 58      * @type {aw.events.EventListener}
 59      * @instance
 60      */
 61     this.addedToParent = this.createListener();
 62     /**
 63      * Executed after component being removed from the parent component.
 64      * @property
 65      * @name aw.components.utils.ComponentInterface#removedFromParent
 66      * @type {aw.events.EventListener}
 67      * @instance
 68      */
 69     this.removedFromParent = this.createListener();
 70   }
 71   aw.components.utils.ComponentInterface = ComponentInterface;
 72   /**
 73    * @type {Object|Function|ComponentInterface}
 74    * @private
 75    */
 76   var p = ComponentInterface.prototype;
 77   /**
 78    * Add child component to this component children list.
 79    * This function is private to other components and must not
 80    * be called. It will be invoked automatically when child
 81    * component registers itself as component instance.
 82    * @function
 83    * @name aw.components.utils.ComponentInterface#$addChild
 84    * @param {aw.components.Component} component
 85    * @instance
 86    */
 87   p.$addChild = function ComponentController_$addChild(component) {
 88     if (component) {
 89       this.children.push(component);
 90       this.childAdded.$fire(component);
 91     }
 92   };
 93   /**
 94    * Remove child component from this component children list.
 95    * This function is private to other components and must not
 96    * be called. It will be invoked automatically when child
 97    * component being removed from component instance registry.
 98    * @function
 99    * @name aw.components.utils.ComponentInterface#$removeChild
100    * @param {aw.components.Component} component
101    * @instance
102    */
103   p.$removeChild = function ComponentController_$removeChild(component) {
104     var index = this.children.indexOf(component);
105     if (component && index >= 0) {
106       this.children.splice(index, 1);
107       this.childRemoved.$fire(component);
108     }
109   };
110   /**
111    * Set component to be a parent to this one.
112    * @function
113    * @name aw.components.utils.ComponentInterface#setParent
114    * @param {aw.components.Component} component
115    * @instance
116    */
117   p.$setParent = function ComponentController_$setParent(component) {
118     if (parent) {
119       this.removedFromParent.$fire(this.parent);
120     }
121     this.parent = component;
122     if (parent) {
123       this.addedToParent.$fire(this.parent);
124     }
125   };
126   /**
127    * Create promise event listener that will be called every time when event fired.
128    * @function
129    * @name aw.components.utils.ComponentInterface#createListener
130    * @returns {aw.events.EventListener}
131    */
132   p.createListener = function Component_createListener() {
133     return aw.events.EventListener.create($scope);
134   };

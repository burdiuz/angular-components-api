 18 (function() {
 19   /**

 22   var utils = window.aw.components.utils;
 23   /**
 24    * Base class for all components controllers
 25    * @class
 26    * @name aw.components.utils.ComponentController
 27    * @constructor
 28    */
 29   function ComponentController() {
 30     var instance = this;
 31     /**
 32      * @function
 33      * @name aw.components.utils.ComponentController#$initialize
 34      * @param {Object} $scope
 35      * @param {aw.components.Component} facade
 36      * @param {Function} preinitHandler
 37      * @instance
 38      */
 39     this.$initialize = function ComponentController_$initialize($scope, facade, preinitHandler) {
 40       if (this.$$initialized) {
 41         throw new Error('Component instance initialized already.');
 42       }
 43       /**
 44        * @type {aw.utils.ValidationHandler}
 45        * @private
 46        */
 47       var refreshValidation = aw.utils.ValidationHandler.create(function () {
 48         //console.log(' - call $digest');
 49         if (!$scope.$$phase) {
 50           //console.log(' -- start $digest');
 51           $scope.$digest();
 52         }
 53       });
 54       /**
 55        * Register watcher to prevent double $digest() calls if component will use $refresh() method
 56        * @type {Function}
 57        */
 58       var digestListenerRemove = $scope.$watch(function () {
 59         //console.log(' --- exec $digest');
 60         refreshValidation.prevent();
 61       });
 62       /*FIXME only if currentTarget === target
 63       $scope.$on('$destroy', function(event){
 64         _destroy.call(instance);
 65         digestListenerRemove();
 66         refreshValidation = null;
 67         componentInterface = null;
 68         instance = null;
 69         facadde = null;
 70       });
 71       var _destroy;
 72       this.$destroy = function(handler){
 73         _destroy = handler;
 74       };
 75       */
 76       /**
 77        * @function
 78        * @name aw.components.utils.ComponentController#$refresh
 79        * Simply forces $scope.$apply()
 80        * Shortcut to if(!$scope.$$phase){ $scope.$apply() };
 81        * @instance
 82        */
 83       this.$refresh = function () {
 84         //console.log('$refresh');
 85         refreshValidation.call();
 86       };
 87       /**
 88        * @function
 89        * @name aw.components.utils.ComponentController#$refreshImmediately
 90        * Simply forces $scope.$apply()
 91        * Shortcut to if(!$scope.$$phase){ $scope.$apply() };
 92        * @instance
 93        */
 94       this.$refreshImmediately = function () {
 95         refreshValidation.callImmediately();
 96       };
 97       /**
 98        * @type {aw.components.utils.ComponentInterface}
 99        * @private
100        */
101       var componentInterface = new aw.components.utils.ComponentInterface();
102       /**
103        * @function
104        * @returns {aw.events.EventListener}
105        */
106       this.$createListener =  componentInterface.createListener;
107       /**
108        * @type {aw.events.EventListener}
109        */
110       this.$childAdded = componentInterface.childAdded;
111       /**
112        * @type {aw.events.EventListener}
113        */
114       this.$childRemoved = componentInterface.childRemoved;
115       /**
116        * @type {aw.events.EventListener}
117        */
118       this.$addedToParent = componentInterface.addedToParent;
119       /**
120        * @type {aw.events.EventListener}
121        */
122       this.$removedFromParent = componentInterface.removedFromParent;
123       /**
124        * @property facade
125        * @memberOf aw.components.utils.ComponentController
126        * @type {Object}
127        * @readOnly
128        * @instance
129        */
130       Object.defineProperty(this, 'facade', {value: facade});
131       /**
132        * @property
133        * @name aw.components.utils.ComponentController#$children
134        * @type {aw.components.utils.Component[]}
135        * @instance
136        */
137       Object.defineProperty(this, '$children', {
138         get: function ComponentController_$children() {
139           return componentInterface.children.slice();
140         }
141       });
142       /**
143        * Parent component
144        * @property $parent
145        * @memberOf aw.components.utils.ComponentController
146        * @type {aw.components.utils.ComponentController}
147        * @instance
148        */
149       Object.defineProperty(this, '$parent', {
150         get: function ComponentController_$parent() {
151           return componentInterface.parent;
152         }
153       });
154       /*
155        Call preinitHandler if setup. Called after initialization but before being added to parent component.
156        */
157       if (typeof(preinitHandler) == "function") {
158         preinitHandler.call(this);
159       }
160       if ($scope) {
161         $scope.$$component = facade;
162         $scope.$$interface = componentInterface;
163       }
164       /**
165        * @property
166        * @name aw.components.utils.ComponentController#$initialized
167        * @type {boolean}
168        * @instance
169        * @private
170        */
171       Object.defineProperty(this, '$initialized', {value: true});
172       /*
173        Register component and find its parent. Starts event sequence $childAdded/$addedToParent if parent found.
174        */
175       aw.components.utils.ComponentScopeRegistry.add($scope, facade, componentInterface);
176     };
177   }

188     return instance instanceof ComponentController;
189   };

193    * @param {Function} constructor
194    * @return {boolean}
195    * @static
196    */
197   ComponentController.isComponentControllerClass = function ComponentController_isComponentControllerClass(constructor) {
198     return constructor instanceof Function && (constructor === ComponentController || ComponentController.prototype.isPrototypeOf(constructor.prototype));
199   };

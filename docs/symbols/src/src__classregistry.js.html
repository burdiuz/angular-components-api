 12 function ClassRegistry(){
 13   /**
 14    *
 15    * @param {function} child
 16    * @param {function} factory
 17    * @param {function} parent
 18    * @param {function} constructionFilter
 19    */
 20   this.mapClass = function(child, factory, parent, constructionFilter){
 21     var index;
 22     if(typeof(child)==="function" && typeof(parent)==="function"){
 23       var id = this.getId(child);
 24       if(id){
 25         throw new Error("Class has registered parent and cannot be extended more. Use mixins instead.");
 26       }
 27       var parentId = this.getId(parent);
 28       if(parentId) parent = getEntry(parentId);
 29       id = setId(child);
 30       setEntry(id, new ClassRegistryEntry(child, factory, parent, constructionFilter));
 31     }
 32   };
 33   /**
 34    *
 35    * @param {function} child
 36    * @param {function} mixin
 37    * @param {function} factory
 38    */
 39   this.mapMixin = function(child, mixin, factory){
 40     // check if class has mixin in ancestors and if has -- throw error
 41     // mixin cannot replace class fields
 42     if(typeof(child)==="function" && typeof(mixin)==="function") {
 43       var id = this.getId(child);
 44       if (id) {
 45         var entry = getEntry(id);
 46         entry.mixin(new ClassRegistryEntry(mixin, factory));
 47       } else {
 48         throw new Error('Class must be registered before adding mixin');
 49       }
 50     }
 51   };
 52   /**
 53    * Instantiate class.
 54    * @param {function} definition
 55    * @param {array} args
 56    */
 57   this.create = function(definition, args){
 58     var id = this.getId(definition);
 59     var instance;
 60     if(id){
 61       var entry = getEntry(id);
 62       instance = entry.create(args || []);
 63     }else{
 64       if(args && args.length) {
 65         var factory = this.makeArgsFactory(args.length);
 66         instance = factory(definition, args);
 67       }else{
 68         instance = new definition();
 69       }
 70     }
 71   };
 72   /**
 73    * Apply parent classes and mixins functionality to already created instance.
 74    * This method is useful when developer does not have control over instance
 75    * creation, so hi may call this method in construction time.
 76    * @param instance
 77    * @param args
 78    * @returns {boolean}
 79    */
 80   this.unpack = function(instance, args){
 81     var definition = instance.prototype.constructor;
 82     var id = this.getId(definition);
 83     if(id){
 84       var entry = getEntry(id);
 85       entry.install(instance, args);
 86       return true;
 87     }
 88     return false;
 89   };
 90   /**
 91    *
 92    * @param {function} definition
 93    * @returns {boolean}
 94    */
 95   this.hasId = function(definition){
 96     return '$__$id__' in definition;
 97   };
 98   /**
 99    *
100    * @param {function} definition
101    * @returns {?number}
102    */
103   this.getId = function(definition){
104     return definition['$__$id__'];
105   };
106   /**
107    *
108    * @param {function} definition
109    * @param {?number} length
110    * @returns {function}
111    */
112   this.makeClassFactory = function(definition, length){
113     length = isNaN(length) ? definition.length :  length;
114     var args = '';
115     if(length>0){
116       while(length-->=0){
117         args = args+',a'+length;
118       }
119       args = args.substr(1);
120     }
121     var constrHandlr;
122     eval('constrHandlr = function('+args+'){ return new definition('+args+');};');
123     return constrHandlr;
124   };
125   /**
126    * Multipurpose factory creator, for unknown class and custom arguments count.
127    * @param length
128    * @returns {function}
129    */
130   this.makeArgsFactory = function(length){
131     if(argMixins[length]) return argMixins[length];
132     var index = length;
133     var args = '';
134     var constrHandlr;
135     if(index>0){
136       while(index-->=0){
137         args = args+',a['+length+']';
138       }
139       args = args.substr(1);
140     }
141     eval('constrHandlr = function(d, a){ return new d('+args+');};');
142     argMixins[length] = constrHandlr;
143     return constrHandlr;
144   };
145   // privates
146   var argMixins = [];
147   var index = 0;
148   var entries = {};
149   function setId(definition){
150     var id = String(++index);
151     definition['$__$id__'] = id;
152     return id;
153   }
154   function setEntry(id, entry){
155     if(id in entries){
156       throw new Error('ClassRegistry entry with id "'+id+'" already exists.');
157     }
158     entries[id] = entry;
159   }
160   function getEntry(id){
161     return entries[id];
162   }
163 }

167   this.definition = child;

170   this.classes = [child];
171   this.mixins = [];
172   this.parent = parent;
173   (function(){
174     var parentClass;
175     if(parent instanceof ClassRegistryEntry){
176       this.classes.push.apply(this.classes, parent.classes);
177       parentClass = parent.definition;
178     }else if(parent){
179       this.classes.push(parent);
180       parentClass = parent;
181     }
182     if(parent) {
183       child.prototype = new parentClass();
184       child.prototype.constructor = child;
185     }
186   })();

188   this.mixin = function(value){
189     if(value instanceof ClassRegistryEntry){
190       this.classes.push.apply(this.classes, value.classes);
191     }else{
192       this.classes.push(value);
193     }
194     this.mixins.push(value);
195   };
196   this.create = function(args){
197     var instance;
198     if(this.factory){
199       instance = this.factory.apply(null, args);
200     }else{

202       instance = factory(this.definition, args);
203     }
204     if(this.parent || this.mixins.length){
205       this.install(instance, args);
206     }
207     return instance;
208   };
